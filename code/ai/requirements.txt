#     pip install audiocraft --no-deps
#     pip install "av==12.0.0" --only-binary=av

#   Step 4 — everything else
#     pip install -r requirements.txt
#
#   Optional TTS backends (install to enable --tts_model xtts / melo):
#     pip install TTS>=0.22.0 --no-deps    # XTTS v2 — prevents torch downgrade to CPU build
#     pip install git+https://github.com/myshell-ai/MeloTTS.git --no-deps  # MeloTTS
#     python -m unidic download            # required post-install step for MeloTTS
#
#   Kokoro language extras (install what you need):
#     pip install "misaki[zh]"         # Chinese TTS
#     pip install "misaki[ja]"         # Japanese TTS

# ---------------------------------------------------------------------------
# PyTorch ecosystem
# MUST install from the PyTorch wheel server — plain PyPI only has CPU builds.
# CPU builds break torchvision (RuntimeError: operator torchvision::nms does not exist).
#
#   pip install torch==2.10.0+cu126 torchvision==0.25.0+cu126 torchaudio==2.10.0+cu126 \
#       --index-url https://download.pytorch.org/whl/cu126
#   pip install xformers --index-url https://download.pytorch.org/whl/cu126
#
# Install MeloTTS and TTS with --no-deps to prevent them downgrading torch to CPU build:
#   pip install git+https://github.com/myshell-ai/MeloTTS.git --no-deps
#   python -m unidic download
#   pip install TTS>=0.22.0 --no-deps   (then install missing deps manually)
#
# CUDA version notes:
#   RTX 4060 driver reports CUDA 13.0; forward-compatible with cu126 builds.
#   cu128 builds also exist but cu126 has the full torch+torchvision+torchaudio trio.
# ---------------------------------------------------------------------------
